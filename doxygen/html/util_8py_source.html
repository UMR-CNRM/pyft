22def debugDecor(func):
23 """
24 Defines a decorator to trace all function calling with arguments and results
25 and count number of calls and time spent
26 """
27 @wraps(func)
28 def wrapper(*args, **kwargs):
29 # Logging call
30 logger = logging.getLogger()
31 if logger.isEnabledFor(logging.DEBUG):
32 callstr = func.__name__ + \
33 '(' + ', '.join([str(a) for a in args] +
34 [k + '=' + str(v) for (k, v) in kwargs.items()]) + ')'
35 logging.debug(callstr + ' --> ...')
36
37 # Count and time
38 if logger.isEnabledFor(logging.INFO):
39 t0 = time.time()
40
41 # effective call
42 result = func(*args, **kwargs)
43
44 # Count and time
45 if logger.isEnabledFor(logging.INFO):
46 from pyft import util
47 if func.__name__ not in util.debugStats:
48 util.debugStats[func.__name__] = dict(nb=0, totalTime=0)
49 util.debugStats[func.__name__]['nb'] += 1
50 duration = time.time() - t0
51 util.debugStats[func.__name__]['totalTime'] += duration
52 util.debugStats[func.__name__]['min'] = \
53 min(duration, util.debugStats[func.__name__].get('min', duration))
54 util.debugStats[func.__name__]['max'] = \
55 max(duration, util.debugStats[func.__name__].get('max', duration))
56
57 # logging result
58 if logger.isEnabledFor(logging.DEBUG):
59 logging.debug(callstr + ' --> ' + str(result))
60
61 return result
62 return wrapper

65def noParallel(func):
66 """
67 Defines a decorator that prevent this method to be executed in parallel on several files
68 """
69 @wraps(func)
70 def wrapper(self, *args, **kwargs):
71 if self.NO_PARALLEL_LOCK is not None:
72 # Acquire lock, if any
73 with self.NO_PARALLEL_LOCK:
74 # We cannot use directly the shared version because the update
75 # tries to send a whole pyft object to the Manager and there are
76 # issues with that (at least about the locks attached to the instance)
77 # The code here allows to synchronize when there is a lock mechanism.
78 # All the routines updating the tree must be decorated by noParallel
79 if self.SHARED_TREE is not None:
80 self.tree.copyFromOtherTree(self.SHARED_TREE)
81 result = func(self, *args, **kwargs)
82 if self.SHARED_TREE is not None:
83 self.tree.copyToOtherTree(self.SHARED_TREE)
84 return result
85 else:
86 return func(self, *args, **kwargs)
87 return wrapper

127def fortran2xml(fortranSource, parser='fxtran', parserOptions=None, wrapH=False):
128 """
129 :param fortranSource: a string containing a fortran source code
130 or a filename
131 :param parser: path to the fxtran parser
132 :param parserOptions: dictionnary holding the parser options
133 :param wrapH: if True, content of .h file is put in a .F90 file (to force
134 fxtran to recognize it as free form) inside a module (to
135 enable the reading of files containing only a code part)
136 :returns: (includesRemoved, xml) where includesRemoved indicates if an include
137 was replaced by fxtran and xml is an ET xml document
138 """
139 # Namespace registration
140 ET.register_namespace('f', NAMESPACE)
141
142 # Default options
143 if parserOptions is None:
144 import pyft
145 parserOptions = pyft.PYFT.DEFAULT_FXTRAN_OPTIONS
146
147 # Call to fxtran
148 renamed = False
149 moduleAdded = False
150 with tempfile.NamedTemporaryFile(buffering=0, suffix='.F90') as file:
151 if os.path.exists(fortranSource):
152 # tempfile not needed in this case if wrapH is False but I found easier to write code
153 # like this to have only one subprocess call and automatic
154 # deletion of the temporary file
155 filename = fortranSource
156 if wrapH and filename.endswith('.h'):
157 renamed = True
158 filename = file.name
159 with open(fortranSource, 'r', encoding='utf-8') as src:
160 content = src.read()
161 # renaming is enough for .h files containing SUBROUTINE or FUNCTION
162 # but if the file contains a code fragment, it must be included in a
163 # program-unit
164 firstLine = [line for line in content.split('\n')
165 if not (re.search(r'^[\t ]*!', line) or
166 re.search(r'^[\t ]*$', line))][0]
167 fisrtLine = firstLine.upper().split()
168 if not ('SUBROUTINE' in fisrtLine or 'FUNCTION' in firstLine):
169 # Needs to be wrapped in a program-unit
170 moduleAdded = True
171 content = 'MODULE FOO\n' + content + '\nEND MODULE FOO'
172 file.write(content.encode('UTF8'))
173 else:
174 filename = file.name
175 file.write(fortranSource.encode('UTF-8'))
176 xml = subprocess.run([parser, filename,
177 '-o', '-'] + parserOptions,
178 stdout=subprocess.PIPE, check=True,
179 encoding='UTF-8').stdout
180 xml = ET.fromstring(xml, parser=ET.XMLParser(encoding='UTF-8'))
181 if renamed:
182 xml.find('./{*}file').attrib['name'] = fortranSource
183 if moduleAdded:
184 file = xml.find('./{*}file')
185 programUnit = file.find('./{*}program-unit')
186 # all nodes inside program-unit except 'MODULE' and 'END MODULE'
187 for node in programUnit[1:-1]:
188 file.append(node)
189 # pylint: disable-next=undefined-loop-variable
190 node.tail = node.tail[:-1] # remove '\n' added before 'END MODULE'
191 file.remove(programUnit)
192
193 includesDone = False
194 if len(set(['-no-include', '-noinclude']).intersection(parserOptions)) == 0:
195 # fxtran has included the files but:
196 # - it doesn't have removed the INCLUDE "file.h" statement
197 # - it included the file with its file node
198 # This code section removes the INCLUDE statement and the file node
199
200 # Remove the include statement
201 includeStmts = xml.findall('.//{*}include')
202 for includeStmt in includeStmts:
203 par = [p for p in xml.iter() if includeStmt in p][0]
204 par.remove(includeStmt)
205 includesDone = True
206 # Remove the file node
207 mainfile = xml.find('./{*}file')
208 for file in mainfile.findall('.//{*}file'):
209 par = [p for p in xml.iter() if file in p][0]
210 index = list(par).index(file)
211 if file.tail is not None:
212 file[-1].tail = file.tail if file[-1].tail is None else (file[-1].tail + file.tail)
213 for node in file[::-1]:
214 par.insert(index, node)
215 par.remove(file)
216
217 return includesDone, xml

228def tofortran(doc):
229 """
230 :param doc: an ET object
231 :return: a string representing the FORTRAN source code
232 """
233 # When fxtran encounters an UTF-8 character, it replaces it by *2* entities
234 # We must first transform each of these entities to its corresponding binary value
235 # (this is done by tostring), then we must consider the result as bytes
236 # to decode these two bytes into UTF-8 (this is done by encode('raw_...').decode('UTF-8'))
237 result = ET.tostring(doc, method='text', encoding='UTF-8').decode('UTF-8')
238 try:
239 result = result.encode('raw_unicode_escape').decode('UTF-8')
240 except UnicodeDecodeError:
241 filename = doc.find('.//{*}file').attrib['name']
242 logging.warning("The file '{}' certainly contains a strange character".format(filename))
243 return result
