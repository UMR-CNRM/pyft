87 def decodeArraySpecs(arraySpecs):
88 asList = []
89 asxList = []
90 for arraySpec in arraySpecs:
91 lb = arraySpec.find('.//{*}lower-bound')
92 ub = arraySpec.find('.//{*}upper-bound')
93 asList.append([alltext(lb) if lb is not None else None,
94 alltext(ub) if ub is not None else None])
95 asxList.append([lb if lb is not None else None, ub if ub is not None else None])
96 return asList, asxList
97
98 result = []
99 for scope in mainScope.getScopes(excludeContains=True):
100 # In case scope is a function, we determine the name of the result
101 if tag(scope[0]) == 'function-stmt':
102 rSpec = scope[0].find('./{*}result-spec/{*}N')
103 funcResultName = rSpec if rSpec is not None else \
104 scope[0].find('./{*}function-N/{*}N')
105 funcResultName = n2name(funcResultName).upper()
106 else:
107 funcResultName = ''
108
109 # Find dummy arguments
110 dummyArgs = [n2name(e).upper() for stmt in scope
111 for e in stmt.findall('.//{*}dummy-arg-LT/{*}arg-N/{*}N')]
112
113 # Loop on each declaration statement
114 for declStmt in [stmt for stmt in scope
115 if tag(stmt) in ('T-decl-stmt' 'component-decl-stmt')]:
116 tSpec = alltext(declStmt.find('.//{*}_T-spec_'))
117 iSpec = declStmt.find('.//{*}intent-spec')
118 if iSpec is not None:
119 iSpec = iSpec.text
120 optSpec = False
121 allocatable = False
122 parameter = False
123 pointer = False
124 allattributes = declStmt.findall('.//{*}attribute/{*}attribute-N')
125 for attribute in allattributes:
126 if alltext(attribute).upper() == 'OPTIONAL':
127 optSpec = True
128 if alltext(attribute).upper() == 'ALLOCATABLE':
129 allocatable = True
130 if alltext(attribute).upper() == 'PARAMETER':
131 parameter = True
132 if alltext(attribute).upper() == 'POINTER':
133 pointer = True
134 # Dimensions declared with the DIMENSION attribute
135 arraySpecs = declStmt.findall('.//{*}attribute//{*}array-spec//{*}shape-spec')
136 as0List, asx0List = decodeArraySpecs(arraySpecs)
137
138 # Loop on each declared variables
139 for enDecl in declStmt.findall('.//{*}EN-decl'):
140 varName = n2name(enDecl.find('.//{*}N')).upper()
141 # Dimensions declared after the variable name
142 arraySpecs = enDecl.findall('.//{*}array-spec//{*}shape-spec')
143 asList, asxList = decodeArraySpecs(arraySpecs)
144 # Initial value (parameter or not)
145 init = enDecl.find('./{*}init-E')
146 if init is not None:
147 init = alltext(init)
148
149 result.append({'as': asList if len(as0List) == 0 else as0List,
150 'asx': asxList if len(asx0List) == 0 else asx0List,
151 'n': varName, 'i': iSpec, 't': tSpec,
152 'arg': varName in dummyArgs,
153 'use': False, 'opt': optSpec, 'allocatable': allocatable,
154 'parameter': parameter, 'pointer': pointer,
155 'result': funcResultName == varName,
156 'init': init, 'scopePath': scope.path})
157
158 # Loop on each use statement
159 for useStmt in [stmt for stmt in scope if tag(stmt) == 'use-stmt']:
160 module = n2name(useStmt.find('.//{*}module-N').find('.//{*}N'))
161 for var in useStmt.findall('.//{*}use-N'):
162 varName = n2name(var.find('.//{*}N'))
163 result.append({'as': None, 'asx': None,
164 'n': varName, 'i': None, 't': None, 'arg': False,
165 'use': module, 'opt': None, 'allocatable': None,
166 'parameter': None, 'pointer': None, 'result': None,
167 'init': None, 'scopePath': scope.path})
168 return result

183 @debugDecor
184 def findVar(self, varName, array=None, exactScope=False, extraVarList=None):
185 """
186 Search for a variable in a list of declared variables
187 :param varName: variable name
188 :param array: True to limit search to arrays,
189 False to limit search to non array,
190 None to return anything
191 :param exactScope: True to limit search to variables declared in the scopePath
192 :param extraVarList: None or list of variables (such as those contained in a VarList object)
193 defined but not yet available in the self.varList object.
194 :return: None if not found or the description of the variable
195
196 The function is designed to return the declaration of a given variable.
197 If we know that the variable is (is not) an array, the last declaration statement
198 must (must not) be an array declaration. If the last declaration statement found doesn't
199 correspond to what is expected, we don't return it.
200 In case array is None, we return the last declaration statement without checking its kind.
201 """
202 extraVarList = extraVarList if extraVarList is not None else []
203 # Select all the variables declared in the current scope or upper,
204 # then select the last declared
205 candidates = {v['scopePath']: v for v in self._fullVarList + extraVarList
206 if v['n'].upper() == varName.upper() and
207 (self._scopePath == v['scopePath'] or
208 (self._scopePath.startswith(v['scopePath'] + '/') and
209 not exactScope))}
210 if len(candidates) > 0:
211 last = candidates[max(candidates, key=len)]
212 if array is True and last.get('as', None) is not None and len(last['as']) > 0:
213 return last
214 if array is False and len(last.get('as', [])) == 0:
215 return last
216 if array is None:
217 return last
218 return None

350 @debugDecor
351 @noParallel
352 @updateVarList
353 @updateTree('signal')
354 def removeVar(self, varList, simplify=False):
355 """
356 :param varList: list of variables to remove. Each item is a list or tuple of two elements.
357 The first one describes where the variable is used, the second one is
358 the name of the variable. The first element is a '/'-separated path with
359 each element having the form 'module:<name of the module>',
360 'sub:<name of the subroutine>', 'func:<name of the function>' or
361 'type:<name of the type>'
362 :param simplify: try to simplify code (if we delete a declaration statement that used a
363 variable as kind selector, and if this variable is not used else where,
364 we also delete it)
365 Remove the variable from declaration, and from the argument list if needed
366 """
367 varList = self._normalizeUniqVar(varList)
368
369 # Sort scopes by depth
370 sortedVarList = {}
371 for scopePath, varName in varList:
372 nb = scopePath.count('/')
373 sortedVarList[nb] = sortedVarList.get(nb, []) + [(scopePath, varName.upper())]
374
375 varToRemoveIfUnused = []
376 # Loop on varList starting by inner most variables
377 nbList = [] if len(sortedVarList.keys()) == 0 else \
378 range(max(sortedVarList.keys()) + 1)[::-1]
379 for nb in nbList:
380 sortedVarList[nb] = sortedVarList.get(nb, [])
381 # Loop on scopes
382 for scopePath in list(set(scopePath for scopePath, _ in sortedVarList[nb])):
383 # use of mainScope because variable can be declared upper than self
384 scope = self.mainScope.getScopeNode(scopePath, excludeContains=True)
385 # Variables searched in this scope
386 varNames = list(set(v for (w, v) in sortedVarList[nb] if w == scopePath))
387 declStmt = _getDeclStmtTag(scopePath)
388 # If scopePath is "module:XX/sub:YY", getScopeNode returns a node
389 # containing the subroutine declaration statements and
390 # excluding the subroutine and functions potentially included
391 # after a "contains" statement
392 previous = None
393 # list() to allow removing during the iteration
394 for node in list(scope):
395 deleted = False
396
397 # Checks if variable is a dummy argument
398 dummyList = node.find('{*}dummy-arg-LT') # This is the list of the dummies
399 if dummyList is not None:
400 # Loop over all dummy arguments
401 for arg in dummyList.findall('.//{*}arg-N'):
402 name = n2name(arg.find('.//{*}N')).upper()
403 for varName in [v for v in varNames if v == name]:
404 # This dummy arg is a searched variable, we remove it from the list
405 scope.removeFromList(arg, dummyList)
406
407 # In case the variable is declared
408 if tag(node) == declStmt:
409 # We are in a declaration statement
410 # list of declaration in the current statment
411 declList = node.find('./{*}EN-decl-LT')
412 for enDecl in declList.findall('.//{*}EN-decl'):
413 name = n2name(enDecl.find('.//{*}N')).upper()
414 for varName in [v for v in varNames if v == name]:
415 # The argument is declared here,
416 # we suppress it from the declaration list
417 varNames.remove(varName)
418 scope.removeFromList(enDecl, declList)
419 # In all the variables are suppressed from the declaration statement
420 if len(list(declList.findall('./{*}EN-decl'))) == 0:
421 if simplify:
422 varToRemoveIfUnused.extend([[scopePath, n2name(nodeN)]
423 for nodeN in node.findall('.//{*}N')])
424 # We will delete the current node but we don't want to lose
425 # any text. So, we put the node's text in the tail of the previous node
426 if previous is not None and node.tail is not None:
427 if previous.tail is None:
428 previous.tail = ''
429 previous.tail += node.tail
430 deleted = True
431 scope.getParent(node).remove(node)
432
433 # In case the variable is a module variable
434 if tag(node) == 'use-stmt':
435 # We are in a use statement
436 useList = node.find('./{*}rename-LT')
437 if useList is not None:
438 for rename in useList.findall('.//{*}rename'):
439 name = n2name(rename.find('.//{*}N')).upper()
440 for varName in [v for v in varNames if v == name]:
441 varNames.remove(varName)
442 # The variable is declared here, we remove it from the list
443 scope.removeFromList(rename, useList)
444 # In case the variable was alone
445 attribute = node.find('{*}module-N').tail
446 if attribute is None:
447 attribute = ''
448 attribute = attribute.replace(' ', '').replace('\n', '')
449 attribute = attribute.replace('&', '').upper()
450 useList = node.find('./{*}rename-LT')
451 if len(useList) == 0 and attribute[0] == ',' and \
452 attribute[1:] == 'ONLY:':
453 # If there is a 'ONLY' attribute,
454 # we suppress the use statement entirely
455 if previous is not None and node.tail is not None:
456 if previous.tail is None:
457 previous.tail = ''
458 previous.tail += node.tail
459 deleted = True
460 scope.getParent(node).remove(node)
461 scope.tree.signal(scope) # Tree must be updated
462 elif len(useList) == 0:
463 # there is no 'ONLY' attribute
464 moduleName = scope.getSiblings(useList, before=True,
465 after=False)[-1]
466 previousTail = moduleName.tail
467 if previousTail is not None:
468 moduleName.tail = previousTail.replace(',', '')
469 scope.getParent(useList).remove(useList)
470 # end loop if all variables have been found
471 if len(varNames) == 0:
472 break
473 # Store node for the following iteration
474 if not deleted:
475 previous = node
476
477 # If some variables have not been found, they are certainly declared one level upper
478 if len(varNames) != 0:
479 newWhere = '/'.join(scopePath.split('/')[:-1])
480 sortedVarList[nb - 1] = sortedVarList.get(nb - 1, []) + \
481 [(newWhere, varName) for varName in varNames]
482
483 if simplify and len(varToRemoveIfUnused) > 0:
484 self.removeVarIfUnused(varToRemoveIfUnused, excludeDummy=True, simplify=True)

486 @debugDecor
487 @updateVarList
488 def addVar(self, varList):
489 """
490 :param varList: list of variable specification to insert in the xml code
491 a variable specification is a list of four elements:
492 - variable scope path (path to module, subroutine, function or type
493 declaration)
494 - variable name
495 - declarative statment
496 - position of the variable in the list of dummy argument,
497 None for a local variable
498 """
499 for (scopePath, name, declStmt, pos) in varList:
500 scope = self.getScopeNode(scopePath)
501
502 # Add variable to the argument list
503 if pos is not None:
504 argN = createElem('arg-N')
505 nodeN = createElem('N')
506 nodeN.append(createElem('n', text=name))
507 argN.append(nodeN)
508 # search for a potential node, within the scope, with a list of dummy arguments
509 argLst = scope.find('.//{*}dummy-arg-LT')
510 if argLst is None:
511 # This was a subroutine or function without dummy arguments
512 scope[0][0].tail = '('
513 argLst = createElem('dummy-arg-LT', tail=')')
514 scope[0].insert(1, argLst)
515 scope.insertInList(pos, argN, argLst)
516
517 # Declare the variable
518 # The following test is needed in case several variables are added in the argument list
519 # but the declaration statement is given only once for all the variables
520 if declStmt is not None and declStmt != '':
521 # Declaration statement tag according to path (member of type declaration or not)
522 declStmtTag = _getDeclStmtTag(scopePath)
523
524 if scopePath.split('/')[-1].split(':')[0] == 'type':
525 # Add declaration statement in type declaration
526 # Statement building
527 ds = createExpr(declStmt)[0]
528 previousTail = '\n' + declStmt[:re.search(r'\S', declStmt).start()]
529
530 # node insertion
531 # scope[0] is the T-stmt node, scope[-1] is the end-T-stmt node
532 # scope[-2] is the last node before the end-T-stmt node (last component,
533 # comment or the T-stmt node)
534 ds.tail = scope[-2].tail
535 scope[-2].tail = previousTail
536 scope.insert(-1, ds) # insert before last one
537
538 else:
539 # Add declaration statement (not type declaration case)
540 # Statement building
541 ds = createExpr(declStmt)[0]
542 previousTail = '\n' + declStmt[:re.search(r'\S', declStmt).start()]
543
544 # node insertion index
545 declLst = [node for node in scope if tag(node) == declStmtTag]
546 if len(declLst) != 0:
547 # There already are declaration statements
548 # We look for the last position in the declaration list which do not use
549 # the variable we add
550 for decl in declLst:
551 index = list(scope).index(decl)
552 if name in [n2name(nodeN) for nodeN in decl.findall('.//{*}N')]:
553 break
554 else:
555 # There is no declaration statement
556 # list of executable nodes
557 stmtLst = [node for node in scope if isExecutable(node)]
558 if len(stmtLst) == 0:
559 # There is no executable statement, we insert the declaration at the end
560 # Last node is the ending node (e.g. end-subroutine-stmt)
561 index = len(scope) - 1
562 else:
563 # We insert the declaration just before the first executable statement
564 index = list(scope).index(stmtLst[0])
565
566 # node insertion
567 if index != 0:
568 ds.tail = scope[index - 1].tail
569 scope[index - 1].tail = previousTail
570 scope.insert(index, ds)

572 @debugDecor
573 @noParallel
574 @updateVarList
575 @updateTree('signal')
576 def addModuleVar(self, moduleVarList):
577 """
578 :param moduleVarList: list of module variable specification to insert in the xml code
579 a module variable specification is a list of three elements:
580 - scope path (path to module, subroutine, function or type
581 declaration)
582 - module name
583 - variable name or or list of variable names
584 or None to add a USE statement without the ONLY attribute
585 For example addModuleVar('sub:FOO', 'MODD_XX', 'Y') will add the following line in
586 subroutine FOO:
587 USE MODD_XX, ONLY: Y
588 """
589 for (scopePath, moduleName, varName) in moduleVarList:
590 if varName is None:
591 varName = []
592 elif not isinstance(varName, list):
593 varName = [varName]
594 scope = self.getScopeNode(scopePath)
595
596 # USE statement already present
597 useLst = [node for node in scope if tag(node) == 'use-stmt']
598
599 # Check if we need to add a USE
600 insertUse = True
601 for us in useLst:
602 usName = n2name(us.find('.//{*}module-N//{*}N'))
603 usVar = [n2name(v.find('.//{*}N')).upper() for v in us.findall('.//{*}use-N')]
604 if len(varName) == 0 and len(usVar) == 0 and usName.upper() == moduleName.upper():
605 # There aleardy is a 'USE MODULE' and we wanted to insert a 'USE MODULE'
606 insertUse = False
607 elif len(varName) > 0 and len(usVar) > 0 and usName.upper() == moduleName.upper():
608 # There already is a 'USE MODULE, ONLY:' and we want to insert another
609 # 'USE MODULE, ONLY:'
610 # We suppress from the varName list, all the variables already defined
611 varName = [var for var in varName if var.upper() not in usVar]
612 if len(varName) == 0:
613 # All the variables we wanted to import are already defined
614 insertUse = False
615
616 if insertUse:
617 # Statement building
618 stmt = f'USE {moduleName}'
619 if len(varName) > 0:
620 stmt += ', ONLY:{}'.format(', '.join(varName))
621 us = createExpr(stmt)[0]
622

806 def modifyAutomaticArrays(self, declTemplate=None, startTemplate=None, endTemplate=None):
807 """
808 :param declTemplate: declaration template
809 :param startTemplate: template for the first executable statement
810 :param: endTemplate: template for the last executable statement
811 :return: number of arrays modified
812 Modifies all automatic arrays declaration in subroutine and functions. The declaration is
813 replaced by the declaration template, the start template is inserted as first executable
814 statement and the end template as last executable statement. Each template can use the
815 following place holders:
816 "{doubledotshape}", "{shape}", "{lowUpList}", "{name}" and "{type}" wich are, respectively
817 modified into ":, :, :", "I, I:J, 0:I", "1, I, I, J, 0, I", "A", "REAL" if the original
818 declaration statement was "A(I, I:J, 0:I)". The template
819 "{type}, DIMENSION({doubledotshape}), ALLOCATABLE :: {name}#
820 ALLOCATE({name}({shape}))#DEALLOCATE({name})"
821 replaces automatic arrays by allocatables
822 """
823 templates = {'decl': declTemplate if declTemplate is not None else '',
824 'start': startTemplate if startTemplate is not None else '',
825 'end': endTemplate if endTemplate is not None else ''} # ordered dict
826
827 number = 0
828 for scope in [scope for scope in self.getScopes(excludeContains=True, includeItself=True)
829 if scope.path.split('/')[-1].split(':')[0] in ('sub', 'func')]:
830 # For all subroutine and function scopes
831 # Determine the list of variables to transform
832 varListToTransform = []
833 for var in [var for var in scope.varList
834 if var['as'] is not None and
835 len(var['as']) > 0 and
836 not (var['arg'] or var['allocatable'] or
837 var['pointer'] or var['result'])]:
838 # For all automatic arrays, which are not argument, not allocatable,
839 # not pointer and not result
840 if var['init'] is not None:
841 logging.warning(("An array ({name}) has an initial value, it can't be " +
842 "processed by modifyAutomaticArrays.)").format(name=var['n']))
843 else:
844 varListToTransform.append(var)
845 # A variable can make use of the size of another variable in its declaration statement
846 # We order the variables to not insert the declaration of a variable before the
847 # declaration of the variables it depends on
848 orderedVarListToTransform = []
849 while len(varListToTransform) > 0:
850 nAdded = 0
851 for var in varListToTransform[:]:
852 # flatten var['asx'] excluding None
853 listN = [x for dim in var['asx'] for x in dim if x is not None]
854 listN = [n2name(nodeN).upper() for asx in listN
855 for nodeN in asx.findall('.//{*}N/{*}n/..')]
856 if len(set(listN).intersection([v['n'].upper()
857 for v in varListToTransform])) == 0:
858 # Variable var does not use variables still in varListToTransform
859 varListToTransform.remove(var)
860 orderedVarListToTransform.append(var)
861 nAdded += 1
862 if nAdded == 0:
863 raise PYFTError('It seems that there is a circular reference in ' +
864 'the declaration statements')
865 # Loop on variable to transform
866 for var in orderedVarListToTransform[::-1]: # reverse order
867 number += 1
868 # Apply the template
869 templ = copy.deepcopy(templates)
870 for templPart in templ:
871 if '{doubledotshape}' in templ[templPart]:
872 templ[templPart] = templ[templPart].replace(
873 '{doubledotshape}', ','.join([':'] * len(var['as'])))
874 if '{shape}' in templ[templPart]:
875 result = []
876 for i in range(len(var['as'])):
877 if var['as'][i][0] is None:
878 result.append(var['as'][i][1])
879 else:
880 result.append(var['as'][i][0] + ':' + var['as'][i][1])
881 templ[templPart] = templ[templPart].replace('{shape}', ', '.join(result))
882 if '{name}' in templ[templPart]:
883 templ[templPart] = templ[templPart].replace('{name}', var['n'])
884 if '{type}' in templ[templPart]:
885 templ[templPart] = templ[templPart].replace('{type}', var['t'])
886 if '{lowUpList}' in templ[templPart]:
887 result = []
888 for i in range(len(var['as'])):
889 if var['as'][i][0] is None:
890 result.extend([1, var['as'][i][1]])
891 else:
892 result.extend([var['as'][i][0], var['as'][i][1]])
893 templ[templPart] = templ[templPart].replace(
894 '{lowUpList}', ', '.join([str(r) for r in result]))
895
896 # Get the xml for the template
897 separator = "!ABCDEFGHIJKLMNOPQRSTUVWabcdefghijklmnopqrstuvwxyz0123456789"
898 part = 0
899 for node in createExpr(templ['decl'] + '\n' + separator + '\n' +
900 templ['start'] + '\n' + separator + '\n' +
901 templ['end']):
902 templPart = list(templ.keys())[part]
903 if not isinstance(templ[templPart], list):
904 templ[templPart] = []
905 if tag(node) == 'C' and node.text == separator:
906 part += 1
907 else:
908 templ[templPart].append(node)
909
910 # Replace declaration statement
911 # We look for the last position in the declaration list which do not use the
912 # variable we add
913 # This algorithm will stop when the original declaration statement is encoutered
914 for decl in scope.findall('./{*}T-decl-stmt'):
915 index = list(scope).index(decl)
916 if var['n'] in [n2name(nodeN) for nodeN in decl.findall('.//{*}N')]:
917 break
918 scope.removeVar([(scope.path, var['n'])], simplify=False)
919 for nnn in templ['decl'][::-1]:
920 scope.insert(index, nnn)
921
922 # Insert statements
923 for nnn in templ['start'][::-1]:
924 scope.insertStatement(nnn, True)
925 for nnn in templ['end'][::-1]:
926 scope.insertStatement(nnn, False)
927 return number
