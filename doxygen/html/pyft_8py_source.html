18@debugDecor
19def conservativePYFT(filename, parser, parserOptions, wrapH,
20 tree=None, verbosity=None, clsPYFT=None):
21 """
22 Return a conservative PYFT object usable for tree manipulation
23 :param filename: name of the file to open
24 :param parser, parserOptions, wrapH: see the pyft class
25 :param tree: Tree instance or None
26 :param verbosity: if not None, sets the verbosity level
27 :param clsPYFT: PYFT class to use
28 :return: PYFT object

52 @updateTree('signal')
53 def __init__(self, filename, output=None, parser=None, parserOptions=None, verbosity=None,
54 wrapH=False, tree=None, enableCache=False):
55 """
56 :param filename: Input file name containing FORTRAN code
57 :param output: Output file name, None to replace input file
58 :param parser: path to the fxtran parser
59 :param parserOptions: dictionnary holding the parser options
60 :param verbosity: if not None, sets the verbosity level
61 :param wrapH: if True, content of .h file is put in a .F90 file (to force
62 fxtran to recognize it as free form) inside a module (to
63 enable the reading of files containing only a code part)
64 :param tree: an optional Tree instance
65 :param enableCache: True to cache node parents
66 """
67 self.__class__.lockFile(filename)
68 if not sys.version_info >= (3, 8):
69 # At least version 3.7 for ordered dictionary
70 # At least verison 3.8 for namsepace wildcard (use of '{*}' in find or findall)
71 raise PYFTError("pyft needs at least version 3.8 of python")
72 self._filename = filename
73 self._originalName = filename
74 assert os.path.exists(filename), 'Input filename must exist'
75 self._output = output
76 self._parser = 'fxtran' if parser is None else parser
77 tree = Tree() if tree is None else tree
78 if self.SHARED_TREE is not None:
79 assert tree is not None, 'tree must be None if setParallel has been called'
80 tree.copyFromOtherTree(self.SHARED_TREE)
81 if parserOptions is None:
82 self._parserOptions = self.DEFAULT_FXTRAN_OPTIONS.copy()
83 else:
84 self._parserOptions = parserOptions.copy()
85 for tDir in tree.getDirs():
86 self._parserOptions.extend(['-I', tDir])
87 for option in self.MANDATORY_FXTRAN_OPTIONS:
88 if option not in self._parserOptions:
89 self._parserOptions.append(option)
90 includesRemoved, xml = fortran2xml(self._filename, self._parser, self._parserOptions, wrapH)
91 super().__init__(xml, enableCache=enableCache, tree=tree)
92 if includesRemoved:
93 self.tree.signal(self)
94 if verbosity is not None:
95 setVerbosity(verbosity)
