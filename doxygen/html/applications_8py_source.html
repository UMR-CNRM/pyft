81 def convertTypesInCompute(self):
82 """
83 Convert STR%VAR into single local variable contained in compute (a-stmt)
84 and in if-then-stmt, else-if-stmt, where-stmt
85 e.g.
86 ZA = 1 + CST%XG ==> ZA = 1 + XCST_G
87 ZA = 1 + PARAM_ICE%XRTMIN(3) ==> ZA = 1 + XPARAM_ICE_XRTMIN3
88 ZRSMIN(1:KRR) = ICED%XRTMIN(1:KRR) => ZRSMIN(1:KRR) = ICEDXRTMIN1KRR(1:KRR)
89 IF(TURBN%CSUBG_MF_PDF=='NONE')THEN => IF(CTURBNSUBG_MF_PDF=='NONE')THEN
90
91 RESTRICTION : works only if the r-component variable is contained in 1 parent structure.
92 Allowed for conversion : CST%XG
93 Not converted : TOTO%CST%XG (for now, recursion must be coded)
94 Not converted : TOTO%ARRAY(:) (shape of the array must be determined from E1)
95 """
96 def convertOneType(component, newVarList, scope):
97 # 1) Build the name of the new variable
98 objType = scope.getParent(component, 2) # The object STR%VAR
99 objTypeStr = alltext(objType)
100 namedENn = objType.find('.//{*}N/{*}n')
101 structure = namedENn.text
102 variable = component.find('.//{*}ct').text
103 # If the variable is an array with index selection
104 # such as ICED%XRTMIN(1:KRR)
105 arrayIndices = ''
106 arrayRall = objType.findall('.//{*}array-R')
107 if len(arrayRall) > 0:
108 arrayR = copy.deepcopy(arrayRall[0]) # Save for the declaration
109 txt = alltext(arrayR).replace(',', '')
110 txt = txt.replace(':', '')
111 txt = txt.replace('(', '')
112 txt = txt.replace(')', '')
113 arrayIndices = arrayIndices + txt
114 elif len(objType.findall('.//{*}element-LT')) > 0:
115 # Case with single element such as ICED%XRTMIN(1)
116 for elem in objType.findall('.//{*}element'):
117 arrayIndices = arrayIndices + alltext(elem)
118 newName = variable[0] + structure + variable[1:] + arrayIndices
119
120 # 2) Replace the namedE>N>n by the newName and delete R-LT
121 # except for array with index selection (R-LT is moved)
122 namedENn.text = newName
123 objType.remove(objType.find('.//{*}R-LT'))
124 if len(arrayRall) > 0:
125 objType.insert(1, arrayR)
126
127 # 3) Add to the list of not already present for declaration
128 if newName not in newVarList:
129 if len(arrayRall) == 0:
130 newVarList[newName] = (None, objTypeStr)
131 else:
132 newVarList[newName] = (arrayR, objTypeStr)
133
134 scopes = self.getScopes()
135 if scopes[0].path.split('/')[-1].split(':')[1][:4] == 'MODD':
136 return
137 for scope in [scope for scope in scopes
138 if 'sub:' in scope.path and 'interface' not in scope.path]:
139 newVarList = {}
140 for ifStmt in (scope.findall('.//{*}if-then-stmt') +
141 scope.findall('.//{*}else-if-stmt') +
142 scope.findall('.//{*}where-stmt')):
143 compo = ifStmt.findall('.//{*}component-R')
144 if len(compo) > 0:
145 for elcompo in compo:
146 convertOneType(elcompo, newVarList, scope)
147
148 for aStmt in scope.findall('.//{*}a-stmt'):
149 # Exclude statements in which the component-R is in E1
150 # (e.g. PARAMI%XRTMIN(4) = 2)
151 if len(aStmt[0].findall('.//{*}component-R')) == 0: # E1 is the first son of aStmt
152 compoE2 = aStmt.findall('.//{*}component-R')
153 if len(compoE2) > 0:
154 # Exclude stmt from which E2 has only 1 named-E/{*}N/{*}n e.g. IKB = D%NKB
155 # warning, it does not handle yet op in simple statement
156 # such as ZEXPL = 1.- TURBN%XIMPL
157 # Include stmt from which E2 has 1 named-E/{*}N/{*}n AND E1 is an array;
158 # e.g. ZDELTVPT(JIJ,JK)=CSTURB%XLINF
159 nbNamedEinE2 = len(aStmt.findall('.//{*}E-2')[0].findall('.//{*}named-E/' +
160 '{*}N/{*}n'))
161 if nbNamedEinE2 > 1 or nbNamedEinE2 == 1 and \
162 len(aStmt[0].findall('.//{*}R-LT')) == 1:
163 for elcompoE2 in compoE2:
164 convertOneType(elcompoE2, newVarList, scope)
165
166 # Add the declaration of the new variables and their affectation
167 for el, var in newVarList.items():
168 if el[0].upper() == 'X' or el[0].upper() == 'P' or el[0].upper() == 'Z':
169 varType = 'REAL'
170 elif el[0].upper() == 'L' or el[0].upper() == 'O':
171 varType = 'LOGICAL'
172 elif el[0].upper() == 'N' or el[0].upper() == 'I' or el[0].upper() == 'K':
173 varType = 'INTEGER'
174 elif el[0].upper() == 'C':
175 varType = 'CHARACTER(LEN=LEN(' + var[1] + '))'
176 else:
177 raise PYFTError('Case not implemented for the first letter of the newVarName' +
178 el + ' in convertTypesInCompute')
179 varArray = ''
180 # Handle the case the variable is an array
181 if var[0]:
182 varArray = ', DIMENSION('
183 for i, sub in enumerate(var[0].findall('.//{*}section-subscript')):
184 if len(sub.findall('.//{*}upper-bound')) > 0:
185 dimSize = simplifyExpr(
186 alltext(sub.findall('.//{*}upper-bound')[0]) +
187 '-' + alltext(sub.findall('.//{*}lower-bound')[0]) +
188 ' + 1')
189 elif len(sub.findall('.//{*}lover-bound')) > 0:
190 dimSize = simplifyExpr(alltext(sub.findall('.//{*}lower-bound')[0]))
191 else: # Case XRTMIN(:)
192 dimSize = 'SIZE(' + var[1] + ',' + str(i+1) + ')'
193 varArray = ', DIMENSION(' + dimSize + ','
194 varArray = varArray[:-1] + ')'
195 scope.addVar([[scope.path, el, varType + varArray + ' :: ' + el, None]])
196
197 # Affectation
198 stmtAffect = createExpr(el + "=" + var[1])[0]
199 scope.insertStatement(scope.indent(stmtAffect), first=True)

264 @debugDecor
265 def addMPPDB_CHECKS(self):
266
267 """
268 Add MPPDB_CHEKS on all intent REAL arrays on subroutines.
269 ****** Not applied on modd_ routines. ********
270 Handle optional arguments.
271 Example, for a BL89 routine with 4 arguments, 1 INTENT(IN),
272 2 INTENT(INOUT), 1 INTENT(OUT), it produces :
273 IF (MPPDB_INITIALIZED) THEN
274 !Check all IN arrays
275 CALL MPPDB_CHECK(PZZ, "BL89 beg:PZZ")
276 !Check all INOUT arrays
277 CALL MPPDB_CHECK(PDZZ, "BL89 beg:PDZZ")
278 CALL MPPDB_CHECK(PTHVREF, "BL89 beg:PTHVREF")
279 END IF
280 ...
281 IF (MPPDB_INITIALIZED) THEN
282 !Check all INOUT arrays
283 CALL MPPDB_CHECK(PDZZ, "BL89 end:PDZZ")
284 CALL MPPDB_CHECK(PTHVREF, "BL89 end:PTHVREF")
285 !Check all OUT arrays
286 CALL MPPDB_CHECK(PLM, "BL89 end:PLM")
287 END IF
288 """
289 def addMPPDB_CHECK_statement(var, subRoutineName, strMSG='beg:'):
290 ifBeg, ifEnd, addD, addLastDim, addSecondDimType = '', '', '', '', ''
291 # Test if the variable is declared with the PHYEX D% structure,
292 # in that case, use the PHYEX MPPDB_CHECK interface
293 if var['as'][0][1]: # If not NoneType
294 if 'D%NIJT' in var['as'][0][1]:
295 addD = 'D,'
296 if len(var['as']) == 2:
297 # This handle 2D arrays with the last dim either D%NKT or anything else.
298 addLastDim = ', ' + var['as'][1][1]
299 if len(var['as']) >= 2:
300 # This adds information on the type of the second dimension :
301 # is it the vertical one or not, to remove extra points
302 if 'D%NK' in var['as'][1][1]:
303 addSecondDimType = ',' + '''"VERTICAL"'''
304 else:
305 addSecondDimType = ',' + '''"OTHER"'''
306 if 'MERGE' in var['as'][-1][1]: # e.g. MERGE(D%NKT,0,OCLOUDMODIFLM)
307 keyDimMerge = var['as'][-1][1].split(',')[2][:-1] # e.g. OCLOUDMODIFLM
308 ifBeg = 'IF (' + keyDimMerge + ') THEN\n'
309 ifEnd = '\nEND IF\n'
310 if var['opt']:
311 ifBeg = ifBeg + 'IF (PRESENT(' + var['n'] + ')) THEN\n IF (SIZE(' + \
312 var['n'] + ',1) > 0) THEN\n'
313 ifEnd = ifEnd + '\nEND IF\nEND IF'
314 argsMPPDB = var['n'] + ", " + "\"" + subRoutineName + " " + strMSG+var['n'] + "\""
315 return createExpr(ifBeg + "CALL MPPDB_CHECK(" + addD + argsMPPDB +
316 addLastDim + addSecondDimType + ")" + ifEnd)[0]
317 scopes = self.getScopes()
318 if scopes[0].path.split('/')[-1].split(':')[1][:4] == 'MODD':
319 return
320 for scope in scopes:
321 # Do not add MPPDB_CHEKS to :
322 # - MODULE or FUNCTION object,
323 # - interface subroutine from a MODI
324 # but only to SUBROUTINES
325 if 'sub:' in scope.path and 'func' not in scope.path and 'interface' not in scope.path:
326 subRoutineName = scope.path.split('/')[-1].split(':')[1]
327
328 # Look for all intent arrays only
329 arraysIn, arraysInOut, arraysOut = [], [], []
330 for var in scope.varList:
331 if var['arg'] and var['as'] and 'TYPE' not in var['t'] and \
332 'REAL' in var['t'] and var['scopePath'] == scope.path:
333 if var['i'] == 'IN':
334 arraysIn.append(var)
335 if var['i'] == 'INOUT':
336 arraysInOut.append(var)
337 if var['i'] == 'OUT':
338 arraysOut.append(var)
339 # Check if there is any intent variables
340 if len(arraysIn) + len(arraysInOut) + len(arraysOut) == 0:
341 break
342
343 # Add necessary module
344 scope.addModuleVar([(scope.path, 'MODE_MPPDB', None)])
345
346 # Prepare some FORTRAN comments
347 commentIN = createElem('C', text='!Check all IN arrays', tail='\n')
348 commentINOUT = createElem('C', text='!Check all INOUT arrays', tail='\n')
349 commentOUT = createElem('C', text='!Check all OUT arrays', tail='\n')
350
351 # 1) variables IN and INOUT block (beggining of the routine)
352 if len(arraysIn) + len(arraysInOut) > 0:
353 ifMPPDBinit = createExpr("IF (MPPDB_INITIALIZED) THEN\n END IF")[0]
354 ifMPPDB = ifMPPDBinit.find('.//{*}if-block')
355
356 # Variables IN
357 if len(arraysIn) > 0:
358 ifMPPDB.insert(1, commentIN)
359 for i, var in enumerate(arraysIn):
360 ifMPPDB.insert(2 + i, addMPPDB_CHECK_statement(var, subRoutineName,
361 strMSG='beg:'))
362
363 # Variables INOUT
364 if len(arraysInOut) > 0:
365 shiftLineNumber = 2 if len(arraysIn) > 0 else 1
366 ifMPPDB.insert(len(arraysIn) + shiftLineNumber, commentINOUT)
367 for i, var in enumerate(arraysInOut):
368 ifMPPDB.insert(len(arraysIn) + shiftLineNumber + 1 + i,
369 addMPPDB_CHECK_statement(var, subRoutineName,
370 strMSG='beg:'))
371
372 # Add the new IN and INOUT block
373 scope.insertStatement(scope.indent(ifMPPDBinit), first=True)
374
375 # 2) variables INOUT and OUT block (end of the routine)
376 if len(arraysInOut) + len(arraysOut) > 0:
377 ifMPPDBend = createExpr("IF (MPPDB_INITIALIZED) THEN\n END IF")[0]
378 ifMPPDB = ifMPPDBend.find('.//{*}if-block')
379
380 # Variables INOUT
381 if len(arraysInOut) > 0:
382 ifMPPDB.insert(1, commentINOUT)
383 for i, var in enumerate(arraysInOut):
384 ifMPPDB.insert(2 + i, addMPPDB_CHECK_statement(var, subRoutineName,
385 strMSG='end:'))
386
387 # Variables OUT
388 if len(arraysOut) > 0:
389 shiftLineNumber = 2 if len(arraysInOut) > 0 else 1
390 ifMPPDB.insert(len(arraysInOut) + shiftLineNumber, commentOUT)
391 for i, var in enumerate(arraysOut):
392 ifMPPDB.insert(len(arraysInOut) + shiftLineNumber + 1 + i,
393 addMPPDB_CHECK_statement(var, subRoutineName,
394 strMSG='end:'))
395
396 # Add the new INOUT and OUT block
397 scope.insertStatement(scope.indent(ifMPPDBend), first=False)

399 @debugDecor
400 def addStack(self, model, stopScopes, parser=None, parserOptions=None, wrapH=False):
401 """
402 Add specific allocations of local arrays on the fly for GPU
403 :param model : 'MESONH' or 'AROME' for specific objects related to the allocator or stack
404 :param stopScopes: scope paths where we stop to add stack
405 :param parser, parserOptions, wrapH: see the pyft class
406
407 Stacks are added to all routines called by the scopes listed in stopScopes
408 """
409 if model == 'AROME':
410 # The AROME transformation needs an additional parameter
411 # We apply the transformation only if the routine is called
412 # from a scope within stopScopes
413 for scope in [scope for scope in self.getScopes()
414 if scope.path in stopScopes or
415 self.tree.isUnderStopScopes(scope.path, stopScopes)]:
416 # Intermediate transformation, needs cpp to be completed
417 # This version would be OK if we didn't need to read again the files with fxtran
418 # after transformation
419 # nb = scope.modifyAutomaticArrays(
420 # declTemplate="temp({type}, {name}, ({shape}))",
421 # startTemplate="alloc({name})")
422
423 # Full transformation, using CRAY pointers
424 # In comparison with the original transformation of Philippe,
425 # we do not call SOF with __FILE__ and __LINE__ because it breaks
426 # future reading with fxtran
427 nb = scope.modifyAutomaticArrays(
428 declTemplate="{type}, DIMENSION({shape}) :: {name}; " +
429 "POINTER(IP_{name}_, {name})",
430 startTemplate="IP_{name}_=YLSTACK%L(KIND({name})/4);" +
431 "YLSTACK%L(KIND({name})/4)=" +
432 "YLSTACK%L(KIND({name})/4)+" +
433 "KIND({name})*SIZE({name});" +
434 "IF(YLSTACK%L(KIND({name})/4)>" +
435 "YLSTACK%U(KIND({name})/4))" +
436 "CALL SOF('" + scope.getFileName() + ":{name}', " +
437 "KIND({name}))")
438
439 if nb > 0:
440 # Some automatic arrays have been modified,
441 # we need to add an argument to the routine
442 scope.addArgInTree('YDSTACK', 'TYPE (STACK) :: YDSTACK', -1,
443 stopScopes, moduleVarList=[('STACK_MOD', ['STACK', 'SOF'])],
444 otherNames=['YLSTACK'],
445 parser=parser, parserOptions=parserOptions, wrapH=wrapH)
446
447 # Copy the stack to a local variable and use it for call statements
448 # this operation must be done after the call to addArgInTree
449 scope.addVar([[scope.path, 'YLSTACK', 'TYPE (STACK) :: YLSTACK', None]])
450 scope.insertStatement(createExpr('YLSTACK=YDSTACK')[0], True)
451 for argN in scope.findall('.//{*}call-stmt/{*}arg-spec/' +
452 '{*}arg/{*}arg-N/../{*}named-E/{*}N'):
453 if n2name(argN) == 'YDSTACK':
454 argN[0].text = 'YLSTACK'
455
456 elif model == 'MESONH':
457 for scope in self.getScopes():
458 # We apply the transformation only if the routine is called
459 # from a scope within stopScopes
460 if (not self.tree.isValid) or stopScopes is None or scope.path in stopScopes or \
461 self.tree.isUnderStopScopes(scope.path, stopScopes):
462 nb = scope.modifyAutomaticArrays(
463 declTemplate="{type}, DIMENSION({doubledotshape}), " +
464 "POINTER, CONTIGUOUS :: {name}",
465 startTemplate="CALL MNH_MEM_GET({name}, {lowUpList})")
466 if nb > 0:
467 # Some automatic arrays have been modified
468 # we need to add the stack module,
469 scope.addModuleVar([(scope.path, 'MODE_MNH_ZWORK',
470 ['MNH_MEM_GET', 'MNH_MEM_POSITION_PIN',
471 'MNH_MEM_RELEASE'])])
472 # to pin the memory position,
473 scope.insertStatement(
474 createExpr(f"CALL MNH_MEM_POSITION_PIN('{scope.path}')")[0], True)
475 # and to realease the memory
476 scope.insertStatement(
477 createExpr(f"CALL MNH_MEM_RELEASE('{scope.path}')")[0], False)
478 else:
479 raise PYFTError('Stack is implemented only for AROME and MESONH models')

503 @debugDecor
504 @updateVarList
505 def removeIJDim(self, stopScopes, parser=None, parserOptions=None, wrapH=False, simplify=False):
506 """
507 Transform routines to be called in a loop on columns
508 :param stopScopes: scope paths where we stop to add the D argument (if needed)
509 :param parser, parserOptions, wrapH: see the pyft class
510 :param simplify: try to simplify code (remove useless dimensions in call)
511
512 ComputeInSingleColumn :
513 - Remove all Do loops on JI and JJ
514 - Initialize former indexes JI, JJ, JIJ to first array element:
515 JI=D%NIB, JJ=D%NJB, JIJ=D%NIJB

517 with explicit (:,*) on CALL statements:
518 e.g. CALL FOO(D, A(:,JK,1), B(:,:))
519 ==> CALL FOO(D, A(JIJ,JK,1), B(:,:)) only if the target argument is not an array
520 """
521
522 indexToCheck = {'JI': ('D%NIB', 'D%NIT'),
523 'JJ': ('D%NJB', 'D%NJT'),
524 'JIJ': ('D%NIJB', 'D%NIJT')}

526 def slice2index(namedE, scope):
527 """
528 Transform a slice on the horizontal dimension into an index
529 Eg.: X(1:D%NIJT, 1:D%NKT) => X(JIJ, 1:D%NKT) Be careful, this array is not contiguous.
530 X(1:D%NIJT, JK) => X(JIJ, JK)
531 :param namedE: array to transform
532 :param scope: scope where the array is
533 """
534 # Loop on all array dimensions
535 for isub, sub in enumerate(namedE.findall('./{*}R-LT/{*}array-R/' +
536 '{*}section-subscript-LT/' +
537 '{*}section-subscript')):
538 if ':' in alltext(sub):
539 loopIndex, _, _ = scope.findIndexArrayBounds(namedE, isub, _loopVarPHYEX)
540 if loopIndex in indexToCheck: # To be transformed
541 if sub.text == ':':
542 sub.text = None
543 lowerBound = createElem('lower-bound')
544 sub.insert(0, lowerBound)
545 else:
546 lowerBound = sub.find('./{*}lower-bound')
547 lowerBound.tail = ''
548 for item in lowerBound:
549 lowerBound.remove(item)
550 upperBound = sub.find('./{*}upper-bound')
551 if upperBound is not None:
552 sub.remove(upperBound)
553 lowerBound.append(createExprPart(loopIndex))
554 if loopIndex not in indexRemoved:
555 indexRemoved.append(loopIndex)
556 # Transform array-R/section-subscript-LT/section-subscript
557 # into parens-R>/element-LT/element if needed
558 if ':' not in alltext(namedE.find('./{*}R-LT/{*}array-R/{*}section-subscript-LT')):
559 namedE.find('./{*}R-LT/{*}array-R').tag = f'{{{NAMESPACE}}}parens-R'
560 namedE.find('./{*}R-LT/{*}parens-R/' +
561 '{*}section-subscript-LT').tag = f'{{{NAMESPACE}}}element-LT'
562 for ss in namedE.findall('./{*}R-LT/{*}parens-R/' +
563 '{*}element-LT/{*}section-subscript'):
564 ss.tag = f'{{{NAMESPACE}}}element'
565 lowerBound = ss.find('./{*}lower-bound')
566 for item in lowerBound:
567 ss.append(item)
568 ss.remove(lowerBound)
569
570 # 0 - Preparation
571 self.addArrayParentheses()
572 self.expandAllArraysPHYEX()
573 if simplify:
574 self.attachArraySpecToEntity()
575 hUupperBounds = [v[1] for v in indexToCheck.values()] # Upper bounds for horizontal dim
576
577 # Loop on all scopes (reversed order); except functions (in particular
578 # FWSED from ice4_sedimentation_stat)
579 for scope in [scope for scope in self.getScopes()[::-1]
580 if 'func:' not in scope.path and
581 (scope.path in stopScopes or
582 self.tree.isUnderStopScopes(scope.path, stopScopes,
583 includeInterfaces=True))]:
584 indexRemoved = []
585
586 # 1 - Remove all DO loops on JI and JJ for preparation to compute on KLEV only
587 # Look for all do-nodes, check if the loop-index is one of the authorized
588 # list (indexToCheck), if found, removes it
589 for doNode in scope.findall('.//{*}do-construct')[::-1]:
590 for loopI in doNode.findall('./{*}do-stmt/{*}do-V/{*}named-E/{*}N'):
591 loopIname = n2name(loopI).upper()
592 if loopIname in indexToCheck:
593 # Move the content of the doNode (except do-stmt and end_do_stmt)
594 # in parent node
595 par = scope.getParent(doNode)
596 index = list(par).index(doNode)
597 for item in doNode[1:-1][::-1]:
598 par.insert(index, item)
599 par.remove(doNode) # remove the do_construct
600 if loopIname not in indexRemoved:
601 indexRemoved.append(loopIname)
602
603 # 2 - Reduce horizontal dimensions for intrinsic array functions
604 # SUM(X(:,:)) => SUM(X(JI, X))
605 # In the simplify==True case, SUM(X(:,:)) becomes SUM(X(:)) by removing first dimension
606 for intr in scope.findall('.//{*}R-LT/{*}parens-R/../..'):
607 intrName = n2name(intr.find('./{*}N')).upper()
608 if intrName in ('PACK', 'UNPACK', 'COUNT', 'MAXVAL', 'MINVAL', 'ALL', 'ANY', 'SUM'):
609 # Is it part of an expression or of an affectation statement?
610 # eg: CALL(UNPACK(Y(:), MASK=G(:,:)) * Z(:))
611 # or X(:,:) = UNPACK(Y(:), MASK=G(:,:)) * Z(:)
612 # If yes, we also need to transform X and Z
613 # if not, only arrays inside the function are transformed
614 parToUse = intr
615 par = intr
616 while par is not None and not isStmt(par):
617 par = scope.getParent(par)
618 if tag(par) in ('a-stmt', 'op-E'):
619 parToUse = par
620
621 # 2.1 Loop on all arrays in the expression using this intrinsic function
622 # to replace horizontal dimensions by indexes
623 for namedE in parToUse.findall('.//{*}R-LT/{*}array-R/../..'):
624 slice2index(namedE, scope)
625
626 # 2.2 Replace intrinsic function when argument becomes a scalar
627 if intr.find('.//{*}R-LT/{*}array-R') is None:
628 if intrName in ('MAXVAL', 'MINVAL', 'SUM', 'ALL', 'ANY'):
629 # eg: MAXVAL(X(:)) => MAXVAL(X(JI)) => X(JI)
630 parens = intr.find('./{*}R-LT/{*}parens-R')
631 parens.tag = f'{{{NAMESPACE}}}parens-E'
632 intrPar = scope.getParent(intr)
633 intrPar.insert(list(intrPar).index(intr), parens)
634 intrPar.remove(intr)
635 elif intrName == 'COUNT':
636 # eg: COUNT(X(:)) => COUNT(X(JI)) => MERGE(1, 0., X(JI))
637 nodeN = intr.find('./{*}N')
638 for item in nodeN[1:]:
639 nodeN.remove(item)
640 nodeN.find('./{*}n').text = 'MERGE'
641 elementLT = intr.find('./{*}R-LT/{*}parens-R/{*}element-LT')
642 for val in (1, 0):
643 element = createElem('element', tail=', ')
644 element.append(createExprPart(val))
645 elementLT.insert(0, element)
646
647 if simplify:
648 # 3 - Remove useless dimensions
649 # Arrays only on horizontal dimensions are transformed into scalars
650 # - at declaration "REAL :: P(D%NIT)" => "REAL :: P"
651 # - during call "CALL FOO(P(:)" => "CALL FOO(P)"
652 # "CALL FOO(Z(:,IK)" => "CALL FOO(Z(JIJ,IK)"
653 # - but "CALL FOO(Z(:,:)" is kept untouched
654 # All arrays are transformed except IN/OUT arrays of the top subroutine (stopScopes)
655 # that cannot be transformed into scalar
656
657 # At least for rain_ice.F90, inlining must be performed before executing this code
658 assert scope.find('.//{*}include') is None and \
659 scope.find('.//{*}include-stmt') is None, \
660 "inlining must be performed before removing horizontal dimensions"
661
662 if scope.path in stopScopes:
663 # List of dummy arguments whose shape cannot be modified
664 preserveShape = [v['n'] for v in scope.varList if v['arg']]
665 else:
666 preserveShape = []
667
668 # 4 - For all subroutines or modi_ interface
669 if 'sub:' in scope.path:
670 # Remove suppressed dimensions "Z(JIJI)" => "Z"
671 # We cannot do this based upon declaration transformation because an array can
672 # be declared in one scope and used in another sub-scope
673 for namedE in scope.findall('.//{*}named-E/{*}R-LT/{*}parens-R/../..'):
674 if n2name(namedE.find('./{*}N')).upper() not in preserveShape:
675 var = scope.varList.findVar(n2name(namedE.find('./{*}N')).upper())
676 if var is not None and var['as'] is not None and len(var['as']) > 0:
677 subs = namedE.findall('./{*}R-LT/{*}parens-R/' +
678 '{*}element-LT/{*}element')
679 if (len(subs) == 1 and var['as'][0][1] in hUupperBounds) or \
680 (len(subs) == 2 and (var['as'][0][1] in hUupperBounds and
681 var['as'][1][1] in hUupperBounds)):
682 namedE.remove(namedE.find('./{*}R-LT'))
683
684 # Remove (:) or (:,:) for horizontal array in call-statement
685 # or replace ':' by index
686 for call in scope.findall('.//{*}call-stmt'):
687 for namedE in call.findall('./{*}arg-spec//{*}named-E'):
688 subs = namedE.findall('.//{*}section-subscript')
689 var = scope.varList.findVar(n2name(namedE.find('./{*}N')).upper())
690 if len(subs) > 0 and (var is None or var['as'] is None or
691 len(var['as']) < len(subs)):
692 # Before adding a warning, functions (especially unpack) must
693 # be recognised
694 # logging.warning(("Don't know if first dimension of {name} must " +
695 # "be modified or not -> kept untouched"
696 # ).format(name=alltext(namedE)))
697 remove = False # to remove completly the parentheses
698 index = False # to transform ':' into index
699 elif (len(subs) >= 2 and
700 ':' in alltext(subs[0]) and var['as'][0][1] in hUupperBounds and
701 ':' in alltext(subs[1]) and var['as'][1][1] in hUupperBounds):
702 # eg: CALL(P(:, :)) with SIZE(P, 1) == D%NIT and SIZE(P, 2) == D%NJT
703 remove = len(subs) == 2
704 index = (len(subs) > 2 and
705 len([sub for sub in subs if ':' in alltext(sub)]) == 2)
706 elif (len(subs) >= 1 and
707 ':' in alltext(subs[0]) and var['as'][0][1] in hUupperBounds):
708 # eg: CALL(P(:)) with SIZE(P, 1) == D%NJT
709 remove = len(subs) == 1
710 index = (len(subs) > 1 and
711 len([sub for sub in subs if ':' in alltext(sub)]) == 1)
712 else:
713 remove = False
714 index = False
715 if remove:
716 if n2name(namedE.find('./{*}N')).upper() in preserveShape:
717 slice2index(namedE, scope)
718 else:
719 nodeRLT = namedE.find('.//{*}R-LT')
720 scope.getParent(nodeRLT).remove(nodeRLT)
721 if index:
722 slice2index(namedE, scope)
723
724 # Remove dimensions in variable declaration statements
725 # This modification must be done after other modifications so that
726 # the findVar method still return an array
727 for decl in scope.findall('.//{*}T-decl-stmt/{*}EN-decl-LT/{*}EN-decl'):
728 name = n2name(decl.find('./{*}EN-N/{*}N')).upper()
729 if name not in preserveShape:
730 varsShape = decl.findall('.//{*}shape-spec-LT')
731 for varShape in varsShape:
732 subs = varShape.findall('.//{*}shape-spec')
733 if (len(subs) == 1 and alltext(subs[0]) in hUupperBounds) or \
734 (len(subs) == 2 and (alltext(subs[0]) in hUupperBounds and
735 alltext(subs[1]) in hUupperBounds)):
736 # Transform array declaration into scalar declaration
737 itemToRemove = scope.getParent(varShape)
738 scope.getParent(itemToRemove).remove(itemToRemove)
739 # We should set scope.varList to None here to clear the cache
740 # but we don't to save some computational time
741
742 # 4 - Values for removed indexes
743 for loopIndex in indexRemoved:
744 # Initialize former indexes JI,JJ,JIJ to first array element:
745 # JI=D%NIB, JJ=D%NJB, JIJ=D%NIJB
746 scope.insertStatement(
747 createExpr(loopIndex + " = " + indexToCheck[loopIndex][0])[0], True)
748 if len(indexRemoved) > 0:
749 scope.addArgInTree('D', 'TYPE(DIMPHYEX_t) :: D',
750 0, stopScopes, moduleVarList=[('MODD_DIMPHYEX', ['DIMPHYEX_t'])],
751 parser=parser, parserOptions=parserOptions, wrapH=wrapH)
752 # Check loop index presence at declaration of the scope
753 scope.addVar([[scope.path, loopIndex, 'INTEGER :: ' + loopIndex, None]
754 for loopIndex in indexRemoved
755 if scope.varList.findVar(loopIndex, exactScope=True) is None])

811 @debugDecor
812 def mathFunctoBRFunc(self):
813 """
814 Convert intrinsic math functions **, LOG, ATAN, **2, **3, **4, EXP, COS, SIN, ATAN2
815 into a self defined function BR_ for MesoNH CPU/GPU bit-reproductibility
816 """
817 # Power integer allowed for BR_Pn and functions converted (from modi_bitrep.f90)
818 powerBRList = [2, 3, 4]
819 mathBRList = ['ALOG', 'LOG', 'EXP', 'COS', 'SIN', 'ASIN', 'ATAN', 'ATAN2']
820
821 for scope in self.getScopes():
822 # 1/2 Look for all operations and seek for power **
823 # <f:op-E>
824 # ... ==> leftOfPow
825 # <f:op>
826 # <f:o>**</f:o>
827 # </f:op>
828 # ... ==> rightOfPow
829 # </f:op-E>
830 for opo in scope.findall('.//{*}o'):
831 if alltext(opo) == '**':
832 op = scope.getParent(opo)
833 opE = scope.getParent(opo, level=2)
834 parOfopE = scope.getParent(opo, level=3)
835 # Save the position of the opE that will be converted
836 index = list(parOfopE).index(opE)
837
838 # Get the next/previous object after/before the ** operator which are
839 # the siblings of the parent of <f:o>*</f:o>
840 rightOfPow = scope.getSiblings(op, after=True, before=False)[0]
841 leftOfPow = scope.getSiblings(op, after=False, before=True)[0]
842
843 # Prepare the object that will contain the left and right of Pow
844 nodeRLT = createElem('R-LT')
845 parensR = createElem('parens-R', text='(', tail=')')
846 elementLT = createElem('element-LT')
847
848 # Regarding the right part of pow(), build a new node expression :
849 # If it is a number and check only for 2, 3 and 4 (e.g. A**2, B**3, D**4 etc)
850 if tag(rightOfPow) == 'literal-E':
851 # Handle '2.' and '2.0'
852 powerNumber = int(alltext(rightOfPow).replace('.', ''))
853 if powerNumber in powerBRList:
854 # <f:named-E>
855 # <f:N>
856 # <f:n>BR_Pn</f:n>
857 # </f:N>
858 # <f:R-LT>
859 # <f:parens-R>(
860 # <f:element-LT>
861 # <f:element>
862 # ... ==> leftOfPow
863 # </f:element>,
864 # </f:element-LT>
865 # </f:parens-R>)
866 # </f:R-LT>
867 # </f:named-E>
868 nodeBRP = createExprPart('BR_P' + str(powerNumber))
869 element = createElem('element')
870 element.append(leftOfPow)
871 elementLT.append(element)
872 # If the right part of pow() is not a number OR it is a number
873 # except 2, 3 or 4 (powerBRList)
874 if tag(rightOfPow) != 'literal-E' or \
875 (tag(rightOfPow) == 'literal-E' and
876 int(alltext(rightOfPow).replace('.', '')) not in powerBRList):
877 # <f:named-E>
878 # <f:N>
879 # <f:n>BR_POW</f:n> or <f:n>BR_Pn</f:n>
880 # </f:N>
881 # <f:R-LT>
882 # <f:parens-R>(
883 # <f:element-LT>
884 # <f:element>
885 # ... ==> leftOfPow
886 # </f:element>,
887 # <f:element>
888 # ... ==> rightOfPow
889 # </f:element>
890 # </f:element-LT>
891 # </f:parens-R>)
892 # </f:R-LT>
893 # </f:named-E>
894 nodeBRP = createExprPart('BR_POW')
895 leftElement = createElem('element', tail=',')
896 leftElement.append(leftOfPow)
897 rightElement = createElem('element')
898 rightElement.append(rightOfPow)
899 elementLT.append(leftElement)
900 elementLT.append(rightElement)
901
902 # Insert the RLT object as a sibling of the BR_ object,
903 # e.g. instead of the old object
904 parensR.append(elementLT)
905 nodeRLT.append(parensR)
906 nodeBRP.insert(1, nodeRLT)
907 nodeBRP.tail = opE.tail
908 parOfopE.remove(opE)
909 parOfopE.insert(index, nodeBRP)
910
911 # Add necessary module in the current scope
912 scope.addModuleVar([(scope.path, 'MODI_BITREP', None)])
913
914 # 2/2 Look for all specific functions LOG, ATAN, EXP,etc
915 for nnn in scope.findall('.//{*}named-E/{*}N/{*}n'):
916 if alltext(nnn).upper() in mathBRList:
917 if alltext(nnn).upper() == 'ALOG':
918 nnn.text = 'BR_LOG'
919 else:
920 nnn.text = 'BR_' + nnn.text
921 # Add necessary module in the current scope
922 scope.addModuleVar([(scope.path, 'MODI_BITREP', None)])

959 def FUNCtoROUTINE(scope, stmt, itemFuncN, localShumansCount, inComputeStmt,
960 nbzshugradwk, zshugradwkDim, dimWorkingVar):
961 """
962 :param scope: node on which the calling function is present before transformation
963 :param stmt: statement node (a-stmt or call-stmt) that contains the function(s) to be
964 transformed
965 :param itemFuncN: <n>FUNCTIONNAME</n> node
966 :param localShumansCount: instance of the shumansGradients dictionnary
967 for the given scope (which contains the number of times a
968 function has been called within a transformation)
969 :param dimWorkingVar: string of the declaration of a potential working variable
970 depending on the array on wich the shuman is applied
971 (e.g. MZM(PRHODJ(:,IKB));
972 dimWorkingVar = 'REAL, DIMENSION(D%NIJT) :: ' )
973 :return zshugradwk
974 :return callStmt: the new CALL to the routines statement
975 :return computeStmt: the a-stmt computation statement if there was an operation
976 in the calling function in stmt
977 """
978 # Function name, parent and grandParent
979 parStmt = scope.getParent(stmt)
980 parItemFuncN = scope.getParent(itemFuncN) # <N><n>MZM</N></n>
981 # <named-E><N><n>MZM</N></n> <R-LT><f:parens-R>(<f:element-LT><f:element>....
982 grandparItemFuncN = scope.getParent(itemFuncN, level=2)
983 funcName = alltext(itemFuncN)
984
985 # workingItem = Content of the function
986 indexForCall = list(parStmt).index(stmt)
987 if inComputeStmt:
988 # one for !$mnh_expand, one for !$acc kernels added at the previous
989 # call to FUNCtoROUTINE
990 indexForCall -= 2
991 siblsItemFuncN = scope.getSiblings(parItemFuncN, after=True, before=False)
992 workingItem = siblsItemFuncN[0][0][0]
993 # Case where & is present in the working item.
994 # We must look for all contents until the last ')'
995 if len(siblsItemFuncN[0][0]) > 1:
996 # last [0] is to avoid getting the '( )' from the function
997 workingItem = scope.updateContinuation(siblsItemFuncN[0][0], removeALL=True,
